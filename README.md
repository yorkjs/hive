# hive

## 安装

CDN

```html
<script src="https://unpkg.com/@yorkjs/hive"></script>
<script>

</script>
```

NPM

```shell
npm install @yorkjs/hive
```

```js
import {
  MS_DAY,
  MS_HOUR,
  MS_MINUTE,
  MS_SECOND,
  MS_WEEK,

  isCustomBarcode,
  isStandardBarcode,

  normalizeVersion,

  moneyToBackend,
  moneyToDisplay,

  rateToBackend,
  rateToDisplay,

  plusNumber,
  minusNumber,
  timesNumber,
  divideNumber,
} from '@yorkjs/hive'
```

```shell
yarn add @yorkjs/hive
```
